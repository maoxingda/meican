## 2. es交互方案查询数据dsl

```json
# 获取doc字段去重结果
GET /dsl_es_idx/_search
{
  "query": {
    "match_all": {}
  },
  "collapse": {
    "field": "department_name"
  },
  "_source": ""
}

# 获取满足特定条件的doc数目
GET /dsl_es_idx/_count
{
  "query": {
    "range": {
      "targeted_date": {
        "gte": "2021-03-01",
        "lte": "2021-03-28",
        "format": "yyyy-MM-dd"
      }
    }
  }
}

# 分页查询、排序、聚合
GET /dsl_es_idx/_search
{
  "from": 0, 
  "size": 20, 
  "query": {
    "range": {
      "targeted_date": {
        "gte": "2021-03-01",
        "lte": "2021-03-28",
        "format": "yyyy-MM-dd"
      }
    }
  },
  "aggs": {
    "total_paid": {
      "sum": {
        "field": "total_paid"
      }
    }
  }, 
  "sort": [
    {
      "targeted_date": {
        "order": "desc"
      }
    },
    {
      "targeted_timestamp": {
        "order": "desc"
      }
    },
    {
      "personal_balance": {
        "order": "desc"
      }
    }
  ]
}

# 索引结构
PUT /dsl_es_idx
{
    "mappings": {

                    }
                }
            },
```
